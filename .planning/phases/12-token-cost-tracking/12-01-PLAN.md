3. In the turn_end handler (around line 139), BEFORE setting status to waiting, add usage accumulation:

```typescript
if (event.type === 'turn_end') {
  // Accumulate token usage if present
  if (event.usage) {
    const msgId = event.usage.message_id;
    // Dedup: skip if we've already counted this message
    if (!msgId || !timers.seenMessageIds.has(msgId)) {
      if (msgId) timers.seenMessageIds.add(msgId);

      // Initialize tokenUsage if not present
      if (!agent.tokenUsage) {
        agent.tokenUsage = {
          inputTokens: 0,
          outputTokens: 0,
          cacheReadTokens: 0,
          cacheCreationTokens: 0,
          costUSD: 0,
        };
      }

      agent.tokenUsage.inputTokens += event.usage.input_tokens;
      agent.tokenUsage.outputTokens += event.usage.output_tokens;
      agent.tokenUsage.cacheReadTokens += (event.usage.cache_read_input_tokens ?? 0);
      agent.tokenUsage.cacheCreationTokens += (event.usage.cache_creation_input_tokens ?? 0);

      // Compute incremental cost for this turn
      agent.tokenUsage.costUSD += computeCost(event.usage, event.usage.model);
    }
  }

  agent.status = 'waiting';
  agent.lastAction = event.content;
  agent.lastUpdated = event.timestamp;
  timers.hadToolsInTurn = false;
  this.emit('agent-update', { ...agent });
  return;
}
```

4. In the `emit('agent-update', { ...agent })` spread, tokenUsage is automatically included because it's on AgentState.
